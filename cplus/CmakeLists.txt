cmake_minimum_required(VERSION 3.12)
project(CppDevEnv VERSION 1.0.0 LANGUAGES CXX)

# Modern C++ Setup
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

# Export compile commands for LSPs (clangd/ccls)
set(CMAKE_EXPORT_COMPILE_COMMANDS ON)

# Dynamic source discovery (re-runs if files are added/deleted)
file(GLOB_RECURSE SRCS CONFIGURE_DEPENDS "src/*.cpp")

# Define target
add_executable(music_app ${SRCS})

# # Target include directories
# target_include_directories(app PRIVATE ${PROJECT_SOURCE_DIR}/include)

# Strict warnings for quality control
if(MSVC)
    target_compile_options(music_app PRIVATE /W4 /WX)
else()
    target_compile_options(music_app PRIVATE -Wall -Wextra -Wpedantic -Werror)
endif()
